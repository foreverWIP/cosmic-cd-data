#alias 0	:	GRAVITY_GROUND

// Collision
#alias 4	:	PLAYER_COL_ROOF


sub ObjectPlayerInteraction
	PlayerObjectCollision(C_BOX, -16, -16, 16, 16)
	if Player.Gravity == GRAVITY_GROUND
		if CheckResult == PLAYER_COL_ROOF
			CallFunction(Player_Kill)
		end if
	end if
end sub

end sub
